\hypertarget{md_docs_2_contributing_autotoc_md83}{}\doxysection{\texorpdfstring{Coding Guidelines}{Coding Guidelines}}\label{md_docs_2_contributing_autotoc_md83}

\begin{DoxyItemize}
\item Prefer clear names and small functions
\item Keep classes focused\+:
\begin{DoxyItemize}
\item shared behavior in {\ttfamily \doxylink{class_base_character}{Base\+Character}}
\item player-\/specific in {\ttfamily \doxylink{class_character}{Character}}
\item enemy-\/specific in {\ttfamily \doxylink{class_enemy}{Enemy}}
\item world objects in {\ttfamily \doxylink{class_prop}{Prop}}
\end{DoxyItemize}
\item Document public APIs with Doxygen comments
\end{DoxyItemize}
